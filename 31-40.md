## Programa 31: Convierte libras y onzas a kilogramos (más precisión usando escala ×1,000,000)
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// Descripción: Convierte libras y onzas a kilogramos (más precisión usando escala ×1,000,000)

.section .rodata
msg: .asciz "150 lb y 8 oz son: "
suf: .asciz " kg (escalado x10^6)\n"

.section .bss
buf: .skip 32

.section .text
.global _start

_start:
    // Datos: 150 lb y 8 oz
    mov x0, #150        // libras
    mov x1, #8          // onzas

    // total_onzas = (150 * 16) + 8 = 2408
    mov x2, #16
    mul x3, x0, x2
    add x3, x3, x1

    // Escalar por 1,000,000 (para precisión)
    ldr x4, =1000000
    mul x5, x3, x4

    // Divisor = 35273968 (≈ 16 * 2.204623 * 1e6)
    ldr x6, =35273968
    udiv x21, x5, x6   // resultado entero truncado pero escalado

    // Mostrar encabezado
    mov x0, #1
    ldr x1, =msg
    mov x2, #21
    mov x8, #64
    svc #0

    // Convertir a texto

## Programa 32: Verifica si un número es divisible entre 2 y/o 5
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// Descripción: Verifica si un número es divisible entre 2 y/o 5

.section .rodata

div5: .asciz "Es divisible entre 5\n"

nope: .asciz "No es divisible entre 2 ni 5\n"

.section .bss
buf: .skip 32

.section .text
.global _start

_start:
    // Mostrar mensaje inicial
    mov x0, #1
    ldr x1, =msg
    mov x2, #20
    mov x8, #64
    svc #0

    // Leer número del usuario
    mov x0, #0
    ldr x1, =buf
    mov x2, #32
    mov x8, #63
    svc #0

    // Convertir ASCII a entero
    ldr x1, =buf
    bl ascii_to_int
    mov x19, x0      // guardar el número

    // Revisar divisibilidad entre 2
    mov x20, #2
    udiv x21, x19, x20
    msub x22, x21, x20, x19
    cmp x22, #0
    cset w23, eq     // w23 = 1 si divisible entre 2

    // Revisar divisibilidad entre 5
    mov x20, #5
    udiv x21, x19, x20
    msub x22, x21, x20, x19
    cmp x22, #0
    cset w24, eq     // w24 = 1 si divisible entre 5

    // Evaluar combinación
    cmp w23, #1
    ccmp w24, #1, #0, eq
    beq ambos

    cmp w23, #1
    beq solo2

    cmp w24, #1
    beq solo5

    b ninguno

ambos:
    ldr x1, =div25
    b mostrar

solo2:
    ldr x1, =div2
    b mostrar

solo5:
    ldr x1, =div5
    b mostrar

ninguno:
    ldr x1, =nope

mostrar:
    mov x0, #1
    mov x2, x1
    mov x3, x1
.cuenta:
    ldrb w4, [x3]
    cmp w4, #0
    beq .escribe
    add x3, x3, #1
    b .cuenta
.escribe:
    sub x2, x3, x2
    mov x8, #64
    svc #0

    // Salir
    mov x8, #93
    mov x0, #0
    svc #0

// -------------------------
// ascii_to_int
// -------------------------
ascii_to_int:
    mov x2, #0
    mov x3, #0
    mov x6, #10
    ldrb w4, [x1]
    cmp w4, #45
    bne .loop_ascii
    mov x3, #1
    add x1, x1, #1
.loop_ascii:
    ldrb w5, [x1], #1
    cmp w5, #10
    beq .fin_ascii
    cmp w5, #13
    beq .fin_ascii
    cmp w5, #0
    beq .fin_ascii
    sub w5, w5, #48
    mul x2, x2, x6
    add x2, x2, w5, uxtw
    b .loop_ascii
.fin_ascii:
    cmp x3, #1
    csel x0, x2, x2, eq
    cbz x3, .ascii_done
    neg x0, x2
