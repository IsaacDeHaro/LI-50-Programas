## Programa 11: Lee 25 enteros y cuenta cuántos son positivos y negativos
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// Descripción: Lee 25 enteros y cuenta cuántos son positivos y negativos

.section .rodata
banner:     .asciz "Ingresa 25 números:\n"
prompt:     .asciz "Número:\n"
msg_pos:    .asciz "Total positivos: "
msg_neg:    .asciz "Total negativos: "
newline:    .asciz "\n"

.section .bss
buffer: .skip 16
outbuf: .skip 32

.section .text
.global _start

_start:
    mov x0, #1
    ldr x1, =banner
    mov x2, #22
    mov x8, #64
    svc #0

    mov x20, #0    // contador general
    mov x21, #0    // positivos
    mov x22, #0    // negativos

loop:
    cmp x20, #25
    beq mostrar_resultado

    mov x0, #1
    ldr x1, =prompt
    mov x2, #8
    mov x8, #64
    svc #0

    mov x0, #0
    ldr x1, =buffer
    mov x2, #16
    mov x8, #63
    svc #0

    ldr x0, =buffer
    bl ascii_to_int

    // Clasificar
    cmp x0, #0
    bge es_positivo
    add x22, x22, #1
    b siguiente

es_positivo:
    add x21, x21, #1

siguiente:
    add x20, x20, #1
    b loop

mostrar_resultado:
    // Positivos
    mov x0, #1
    ldr x1, =msg_pos
    mov x2, #18
    mov x8, #64
    svc #0

    mov x0, x21
    ldr x1, =outbuf
    bl int_to_ascii
    mov x2, #16
    mov x0, #1
    mov x8, #64
    svc #0

    // Nueva línea
    mov x0, #1
    ldr x1, =newline
    mov x2, #1
    mov x8, #64
    svc #0

    // Negativos
    mov x0, #1
    ldr x1, =msg_neg
    mov x2, #18
    mov x8, #64
    svc #0

    mov x0, x22
    ldr x1, =outbuf
    bl int_to_ascii
    mov x2, #16
    mov x0, #1
    mov x8, #64
    svc #0

    mov x0, #1
    ldr x1, =newline
    mov x2, #1
    mov x8, #64
    svc #0

    mov x8, #93
    mov x0, #0
    svc #0

// Subrutinas
ascii_to_int:
    mov x1, #0
    mov x4, #10
    ldrb w3, [x0]
    cmp w3, #45      // ¿negativo?
    bne .convert
    add x0, x0, #1
    bl .convert
    neg x0, x0
    ret

.convert:
    mov x1, #0
.loop:
    ldrb w2, [x0], #1
    cmp w2, #10
    beq .done
    sub w2, w2, #48
    mul x1, x1, x4
    add x1, x1, x2
    b .loop
.done:
    mov x0, x1
    ret

int_to_ascii:
    mov x2, x1
    mov x3, #10
    mov x4, x0
    add x2, x2, #15
    mov w5, #10
    strb w5, [x2]
    sub x2, x2, #1
    mov w5, #0
    strb w5, [x2]
.itoa_loop:
    udiv x5, x4, x3
    msub x6, x5, x3, x4
    add x6, x6, #48
    strb w6, [x2]
    mov x4, x5
    sub x2, x2, #1
    cbz x5, .itoa_end
    b .itoa_loop
.itoa_end:
    add x1, x2, #1
    ret
```

## Programa 15: Imprime tabla de suma del 1 al 12 en formato i + j = r
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// Descripción: Imprime tabla de suma del 1 al 12 en formato i + j = r

.section .rodata
banner: .asciz "Tabla de suma del 1 al 12:\n"
plus:   .asciz " + "
eq:     .asciz " = "
nl:     .asciz "\n"

.section .bss
buffer: .skip 16

.section .text
.global _start

_start:
    // Mostrar encabezado
    mov x0, #1
    ldr x1, =banner
    mov x2, #27
    mov x8, #64
    svc #0

    mov x19, #1      // i

fila:
    cmp x19, #13
    bge fin

    mov x20, #1      // j

col:
    cmp x20, #13
    bge sig_fila

    // Mostrar i
    mov x0, x19
    ldr x1, =buffer
    bl int_to_ascii
    mov x0, #1
    mov x2, #16
    mov x8, #64
    svc #0

    // Mostrar " + "
    mov x0, #1
    ldr x1, =plus
    mov x2, #3
    mov x8, #64
    svc #0

    // Mostrar j
    mov x0, x20
    ldr x1, =buffer
    bl int_to_ascii
    mov x0, #1
    mov x2, #16
    mov x8, #64
    svc #0

    // Mostrar " = "
    mov x0, #1
    ldr x1, =eq
    mov x2, #3
    mov x8, #64
    svc #0

    // Mostrar resultado (i + j)
    add x0, x19, x20
    ldr x1, =buffer
    bl int_to_ascii
    mov x0, #1
    mov x2, #16
    mov x8, #64
    svc #0

    mov x8, #64

fin:
```

```
msg2:    .asciz "Ingresa el segundo número: "

    // Mensaje 1

    mov x8, #64
    svc #0

    // Leer primer número
    mov x0, #0
    ldr x1, =buf
    mov x2, #16
    mov x8, #63
    svc #0

    b fin

.negativo:
    mov x0, #1
    ldr x1, =res_neg
    mov x2, #22
    mov x8, #64
    svc #0

fin:
    mov x8, #93
    mov x0, #0
    svc #0

// ascii_to_int: x1 = puntero al string → x0 = resultado entero con signo
ascii_to_int:
    mov x2, #0     // acumulador
    mov x3, #0     // signo
    mov x6, #10    // base decimal

    ldrb w4, [x1]
    cmp w4, #45     // '-'
    bne .loop
    mov x3, #1
    add x1, x1, #1

.loop:
    ldrb w5, [x1], #1
    cmp w5, #10     // newline
    beq .fin
    cmp w5, #13     // carriage return
    beq .fin
    cmp w5, #0
    beq .fin
    sub w5, w5, #48
    mul x2, x2, x6
    add x2, x2, w5, uxtw
    b .loop

.fin:
    cmp x3, #1
    csel x0, x2, x2, eq
    cbz x3, .fin_done
    neg x0, x2
.fin_done:
    ret
```
