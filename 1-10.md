## Programa 1: Imprimir los enteros del 9 al 43
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// C: for (int i = 9; i <= 43; i++) printf("%d\n", i)

.section .data
buffer: .skip 12        // Espacio para número + '\n' + '\0'

.section .text
.global _start

_start:
    mov x19, #9         // x19 = i = 9

loop1:
    cmp x19, #43
    bgt end1

    // Convertir x19 a cadena
    mov x0, x19
    ldr x1, =buffer
    bl int_to_ascii

    // Imprimir
    mov x8, #64         // syscall write
    mov x0, #1          // stdout
    ldr x1, =buffer
    mov x2, #12
    svc #0

    add x19, x19, #1
    b loop1

end1:
    mov x8, #93         // syscall exit
    mov x0, #0
    svc #0

// ----------------------------
// subrutina: int_to_ascii
// entrada: x0 = número
// salida: buffer con ascii en x1
// ----------------------------
int_to_ascii:
    mov x2, x1          // puntero al buffer
    mov x3, #10         // divisor
    mov x4, x0          // copia del número
    add x2, x2, #11     // apunta al final
    mov w5, #10
    strb w5, [x2]       // '\n'
    sub x2, x2, #1
    mov w5, #0
    strb w5, [x2]       // '\0'

itoa_loop:
    udiv x5, x4, x3
    msub x6, x5, x3, x4 // x6 = x4 - x5*x3 (resto)
    add x6, x6, #48     // convierte a ascii
    strb w6, [x2]
    mov x4, x5
    sub x2, x2, #1
    cbz x5, itoa_end
    b itoa_loop

itoa_end:
    add x1, x2, #1
    ret
```

## Programa 4: Imprime enteros del 1 al 30 junto con su recíproco (n = 1/n)
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// Descripción: Imprime enteros del 1 al 30 junto con su recíproco (n = 1/n)

.section .rodata
banner: .asciz "Mostrando enteros del 1 al 30 con su recíproco:\n"

.section .bss
buffer_num: .skip 16     // Para guardar número convertido
line:       .skip 64     // Para construir cada línea completa

.section .text
.global _start

_start:
    // Imprimir encabezado
    mov x0, #1
    ldr x1, =banner
    mov x2, #43
    mov x8, #64
    svc #0

    mov x19, #1              // i = 1

loop:
    cmp x19, #30
    bgt fin

    // Convertir número a ASCII → buffer_num
    mov x0, x19
    ldr x1, =buffer_num
    bl int_to_ascii

    // Iniciar ptr de escritura en línea
    ldr x2, =line            // x2 = ptr_linea
    ldr x3, =buffer_num      // x3 = ptr_num
copiar_num:
    ldrb w4, [x3], #1
    strb w4, [x2], #1
    cbz w4, fin_copia
    b copiar_num

fin_copia:
    // Agregar cadena " = 1/"
    mov w4, #' '
    strb w4, [x2], #1
    mov w4, #'='
    strb w4, [x2], #1
    mov w4, #' '
    strb w4, [x2], #1
    mov w4, #'1'
    strb w4, [x2], #1
    mov w4, #'/'
    strb w4, [x2], #1

    // Reconvertir número a ASCII otra vez
    mov x0, x19
    mov x1, x2
    bl int_to_ascii

    // Buscar fin para meter '\n'
buscar_fin:
    ldrb w5, [x1], #1
    cbz w5, meter_nl
    b buscar_fin

meter_nl:
    mov w5, #10
    strb w5, [x1]

    // Imprimir línea
    ldr x1, =line
    mov x0, #1
    mov x2, #64
    mov x8, #64
    svc #0

    add x19, x19, #1
    b loop

fin:
    mov x8, #93
    mov x0, #0
    svc #0

// --------------------------------------
// Convierte x0 (entero) a ASCII en x1 (puntero a buffer)
// Devuelve en x1 el inicio de la cadena
// --------------------------------------
int_to_ascii:
    mov x2, x1          // ptr
    mov x3, #10         // divisor base 10
    mov x4, x0          // número a convertir
    add x2, x2, #15     // final del buffer
    mov w5, #0
    strb w5, [x2]
    sub x2, x2, #1

itoa_loop:
    udiv x5, x4, x3
    msub x6, x5, x3, x4
    add x6, x6, #48
    strb w6, [x2]
    mov x4, x5
    sub x2, x2, #1
    cbz x5, itoa_end
    b itoa_loop

itoa_end:
    add x1, x2, #1
    ret
```

## Programa 7: Pide al usuario un número y verifica si es divisible entre 14
```
// Autor: De Haro Ramirez Isaac
// Fecha: 9 de abril de 2025
// Descripción: Pide al usuario un número y verifica si es divisible entre 14

.section .rodata
banner:  .asciz "Ingresa un número para verificar si es divisible entre 14:\n"
msg_yes: .asciz "Sí es divisible entre 14\n"
msg_no:  .asciz "No es divisible entre 14\n"

.section .bss
buffer: .skip 16       // Espacio para leer número

.section .text
.global _start

_start:
    // Mostrar banner
    mov x0, #1
    ldr x1, =banner
    mov x2, #52
    mov x8, #64
    svc #0

    // Leer número (máx 16 bytes desde teclado)
    mov x0, #0
    ldr x1, =buffer
    mov x2, #16
    mov x8, #63         // syscall read
    svc #0

    // Convertir ASCII a número entero
    ldr x0, =buffer
    bl ascii_to_int     // devuelve entero en x0

    // Verificar divisibilidad entre 14
    mov x1, #14
    udiv x2, x0, x1
    msub x3, x2, x1, x0   // x3 = x0 - x2 * x1

    cbz x3, divisible

    // No divisible
    mov x0, #1
    ldr x1, =msg_no
    mov x2, #26
    mov x8, #64
    svc #0
    b fin

divisible:
    mov x0, #1
    ldr x1, =msg_yes
    mov x2, #24
    mov x8, #64
    svc #0

fin:
    mov x8, #93
    mov x0, #0
    svc #0

// Subrutina: ascii_to_int
// Entrada: x0 = ptr a string ASCII
// Salida: x0 = número como entero
ascii_to_int:
    mov x1, #0          // acumulador
    mov x4, #10         // multiplicador constante

.next_char:
    ldrb w2, [x0], #1
    cmp w2, #10         // ¿Enter? \n
    beq .done
    sub w2, w2, #48     // ASCII → número
    mul x1, x1, x4
    add x1, x1, x2
    b .next_char

.done:
    mov x0, x1
    ret
```
